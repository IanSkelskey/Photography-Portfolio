import Head from 'next/head';
import { useState } from 'react'

import TopNav from '../components/TopNav';
import NavDrawer from '../components/NavDrawer';
import NavDrawerItem from '../components/NavDrawerItem';

import { AiOutlineHeart, AiFillFileText, AiFillCamera, AiFillHome, AiFillInfoCircle } from 'react-icons/ai';
import { RiGalleryFill } from 'react-icons/ri'

/**
 * 
 * @param pageComponent The page component to be rendered.
 * @param title The title of the page component. 
 * @returns The provided page component plus a navigation bar and some meta data.
 */
export default function ContentPage({ pageComponent, title }) {

    const [drawerIsOpen, setDrawerIsOpen] = useState(false);

    return (
        <div className='min-h-screen'>
            <Head>
                <title>{title}</title>
                <meta
                    name="description"
                    content="A photography portfolio for web generated by create next app. Uses TailwindCSS." />
                <link rel="icon" href="/favicon.ico" />
            </Head>

            <main>
                <TopNav
                    title={'Your Name'}
                    onBurgerMenuClicked={() => setDrawerIsOpen(!drawerIsOpen)}
                    subcomponents={
                        [
                            <a key={2} className='font-medium' href='#'>Contact</a>,
                            <AiOutlineHeart key={3} className='text-2xl' />
                        ]
                    } />

                {/** Main Content */}
                <div>

                    {drawerIsOpen ?
                        <NavDrawer
                            items={
                                [
                                    <NavDrawerItem
                                        key={0}
                                        icon={<AiFillHome />}
                                        target='/'
                                        text='Home'
                                    />,
                                    <NavDrawerItem
                                        key={1}
                                        icon={<AiFillInfoCircle />}
                                        target='/About'
                                        text='About'
                                    />,
                                    <NavDrawerItem
                                        key={2}
                                        icon={<RiGalleryFill />}
                                        target='/Gallery'
                                        text='Gallery'
                                    />,
                                    <NavDrawerItem
                                        key={3}
                                        icon={<AiFillFileText />}
                                        target='#'
                                        text='Resume'
                                    />,
                                    <NavDrawerItem
                                        key={4}
                                        icon={<AiFillCamera />}
                                        target='/#services'
                                        text='Services'
                                    />
                                ]
                            }
                        />
                        : null}
                </div>

                {pageComponent}

            </main>

        </div>
    );
}
